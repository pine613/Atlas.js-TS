他にもAtlas.jsには便利なプロパティ、メソッドがいろいろ用意されています
それらを紹介しましょう 
var util = new Util();
util.isMobile;
/*
これはプロパティですが
クライアントが使用している機器が携帯端末か
どうかを判定します。
携帯端末ならtrue,そうでなければfalseを返します。
 */
game.drawText(x, y, string, col,font);
/*
座標(x,y)に文字列"string",色"col",フォント"font"で
書き込みます。
*/
game.drawBox(x, y, sizeX, sizeY, col, alpha);
座標(x,y)に横"sizeX"px,縦"sizeY"pxで色"col",透明度"alpha"の四角形を
書き込みます
*/
game.getRand(min,max);
/*
minからmaxの間の適当な整数を返します。
*/
game.field;
/*
game化されたcanvas要素が入っています。
ここからcanvas要素を操作することが出来ます。
例
game.field.width = 500;
ゲーム画面の幅を500に設定する
ただし、例の場合は、画面サイズ変更のメソッドとして
game.changeSize(width,height);
があり、こちらを使うことを推奨します。
*/
さらに、Atlas.jsにはクラスを継承し、サブクラスを作るメソッドが用意されています

例

Enemyクラスの生成

(function(){
    Enemy = Atlas.createClass(Atlas.Tile,{
	    initialize : function(image,numX,numY){
		    this.inherent(image,numX,numY);
		    this.sx = this.getRand(5,10);
			this.sy = this.getRand(5,10);
			this.setPosition(0,this.getRand(100,300));
		},
		move : function(bullets){
		    this.drawGraph();
		    this.x += this.sx;
			this.y += this.sy;			
			if(this.x > 500)
			    this.remove = true;
		}
	});
});
window.onload = function(){
    Atlas("game");
    var game = new Screen();
    var enemy = new Enemy("teki2.png",100,95);
    game.mainScene(function(){
        enemy.move();
    });
};

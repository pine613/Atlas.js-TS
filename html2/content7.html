in Addition,you can override the Atlas's Classes and create your new Classes

Example(Creating Enemy Class)

(function(){
    Enemy = Atlas.createClass(Atlas.Tile,{
	    initialize : function(image,numX,numY){
		    this.inherent(image,numX,numY);
		    this.sx = this.getRand(5,10);
			this.sy = this.getRand(5,10);
			this.setPosition(0,this.getRand(100,300));
		},
		move : function(){
		    this.drawGraph();
		    this.x += this.sx;
			this.y += this.sy;			
			if(this.x > 500)
			    this.remove = true;
		}
	});
});
window.onload = function(){
    Atlas("game");
    var game = new Screen();
    var enemy = new Enemy("teki2.png",100,95);
    game.mainScene(function(){
        enemy.move();
    });
    game.start();
};
